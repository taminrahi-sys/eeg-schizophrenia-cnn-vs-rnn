def preprocess_signal(signal_data, fs=250):
    """Vorverarbeitung des EEG-Signals"""
    # 1. Bandpass-Filter (0.5-50 Hz)
    nyquist = fs / 2
    low = 0.5 / nyquist
    high = 50 / nyquist
    b, a = signal.butter(4, [low, high], btype='band')
    filtered_signal = signal.filtfilt(b, a, signal_data)
    
    # 2. Notch-Filter für 50 Hz (Netzfrequenz)
    notch_freq = 50.0 / nyquist
    b_notch, a_notch = signal.iirnotch(notch_freq, Q=30)
    filtered_signal = signal.filtfilt(b_notch, a_notch, filtered_signal)
    
    return filtered_signal

def extract_features(signal_data, fs=250):
    """Extrahiert umfangreiche Features aus dem EEG-Signal"""
    features = []
    
    # Zeit-Domäne Features
    features.append(np.mean(signal_data))  # Mean
    features.append(np.std(signal_data))   # Standard deviation
    features.append(np.var(signal_data))   # Variance
    features.append(stats.skew(signal_data))  # Skewness
    features.append(stats.kurtosis(signal_data))  # Kurtosis
    features.append(np.max(signal_data))   # Maximum
    features.append(np.min(signal_data))   # Minimum
    features.append(np.ptp(signal_data))   # Peak-to-peak
    features.append(np.median(signal_data))  # Median
    features.append(np.percentile(signal_data, 25))  # 25th percentile
    features.append(np.percentile(signal_data, 75))  # 75th percentile
    
    # RMS (Root Mean Square)
    features.append(np.sqrt(np.mean(signal_data**2)))
    
    # Zero-Crossing Rate
    zero_crossings = np.sum(np.diff(np.sign(signal_data)) != 0)
    features.append(zero_crossings / len(signal_data))
    
    # Frequenz-Domäne Features (FFT)
    N = len(signal_data)
    fft_vals = np.abs(fft(signal_data)[:N//2])
    fft_freqs = fftfreq(N, 1/fs)[:N//2]
    
    # Power in verschiedenen Frequenzbändern
    # Delta (0.5-4 Hz)
    delta_mask = (fft_freqs >= 0.5) & (fft_freqs < 4)
    features.append(np.sum(fft_vals[delta_mask]**2))
    
    # Theta (4-8 Hz)
    theta_mask = (fft_freqs >= 4) & (fft_freqs < 8)
    features.append(np.sum(fft_vals[theta_mask]**2))
    
    # Alpha (8-13 Hz)
    alpha_mask = (fft_freqs >= 8) & (fft_freqs < 13)
    features.append(np.sum(fft_vals[alpha_mask]**2))
    
    # Beta (13-30 Hz)
    beta_mask = (fft_freqs >= 13) & (fft_freqs < 30)
    features.append(np.sum(fft_vals[beta_mask]**2))
    
    # Gamma (30-50 Hz)
    gamma_mask = (fft_freqs >= 30) & (fft_freqs <= 50)
    features.append(np.sum(fft_vals[gamma_mask]**2))
    
    # Spektrale Features
    features.append(np.mean(fft_vals))  # Mean spektrale Amplitude
    features.append(np.std(fft_vals))   # Std spektrale Amplitude
    features.append(np.max(fft_vals))   # Peak spektrale Amplitude
    
    # Spektraler Schwerpunkt
    spectral_centroid = np.sum(fft_freqs * fft_vals) / np.sum(fft_vals)
    features.append(spectral_centroid)
    
    # Statistische Features der ersten Differenzen
    diff_signal = np.diff(signal_data)
    features.append(np.mean(np.abs(diff_signal)))  # Mean absolute difference
    features.append(np.std(diff_signal))  # Std of differences
    
    return np.array(features)

In [12]:
# Grid Search für beste Hyperparameter
param_grid = {
    'C': [0.1, 1, 10, 100],
    'gamma': ['scale', 'auto', 0.001, 0.01, 0.1],
    'kernel': ['rbf', 'linear', 'poly']
}

print("Starte Grid Search für optimale Hyperparameter...")
print("Dies kann einige Minuten dauern...\n")

# Stratified K-Fold Cross-Validation
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

# Grid Search
grid_search = GridSearchCV(
    SVC(random_state=42),
    param_grid,
    cv=cv,
    scoring='f1',
    n_jobs=-1,
    verbose=2
)

grid_search.fit(X_train_scaled, y_train)

print(f"\nBeste Parameter: {grid_search.best_params_}")
print(f"Bester CV F1-Score: {grid_search.best_score_:.4f}")
