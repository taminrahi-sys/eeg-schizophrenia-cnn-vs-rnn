# Zelle 2: Loader & Vorverarbeitung (EEG -> Epochen)
def load_eea_file(path):
    """
    TODO: echten .eea-Parser implementieren.
    Rückgabe: np.ndarray float32, shape (n_channels, n_samples)
    """
    rng = np.random.default_rng(abs(hash(path)) % (2**32))
    n_channels, n_samples = 32, 10_000
    return rng.normal(0, 1, size=(n_channels, n_samples)).astype(np.float32)

def bandpass_filter(eeg, fs=FS, low=1, high=40, order=4):
    ny = 0.5 * fs
    b, a = butter(order, [low/ny, high/ny], btype='band')
    return filtfilt(b, a, eeg, axis=1)

def zscore_per_channel(eeg, eps=1e-8):
    m = eeg.mean(axis=1, keepdims=True)
    s = eeg.std(axis=1, keepdims=True)
    return (eeg - m) / (s + eps)

def clip_artifacts(eeg, thresh_uv=THRESH_UV):
    return eeg if thresh_uv is None else np.clip(eeg, -thresh_uv, thresh_uv)

def segment_epochs(eeg, fs=FS, epoch_len_s=EPOCH_LEN_S, step_s=STEP_S):
    # Input (C, T) -> Output (E, T_win, C)
    win  = int(epoch_len_s * fs)
    step = int(step_s * fs)
    C, T = eeg.shape
    if T < win: return np.empty((0, win, C), dtype=eeg.dtype)
    segs = []
    for s in range(0, T - win + 1, step):
        segs.append(eeg[:, s:s+win].T)  # (T_win, C)
    return np.stack(segs, axis=0)

# Zelle 3: STFT -> Log-Spektrogramm (mit tf.boolean_mask statt Bool-Indexing)
def to_log_stft(x):
    """
    x: tf.Tensor (T, C) float32
    Rückgabe: tf.Tensor (frames, freq_kept, C) float32
    """
    x = tf.convert_to_tensor(x, dtype=tf.float32)   # (T, C)
    xc = tf.transpose(x, [1, 0])                    # (C, T)

    spec_c = tf.signal.stft(
        xc, frame_length=N_FFT, frame_step=HOP, fft_length=N_FFT,
        window_fn=tf.signal.hann_window
    )                                              # (C, frames, fft_bins)
    spec_c = tf.abs(spec_c) + 1e-7

    spec = tf.transpose(spec_c, [1, 2, 0])         # (frames, fft_bins, C)

    n_bins = tf.shape(spec)[1]
    freqs = tf.linspace(0.0, float(FS) / 2.0, n_bins)  # (fft_bins,)
    mask  = freqs <= tf.cast(F_CUT, tf.float32)
    spec  = tf.boolean_mask(spec, mask, axis=1)        # (frames, kept_bins, C)

    spec  = tf.math.log(spec)
    return spec

# Zelle 4: Dataset mit Gruppen (Dateiname) – gegen Leakage
def build_dataset_with_groups(data_dir):
    Xs, ys, gs = [], [], []
    for cls_name, cls_label in CLASS_MAP.items():
        folder = os.path.join(data_dir, cls_name)
        files  = sorted(glob.glob(os.path.join(folder, "*.eea")))
        for f in files:
            raw = load_eea_file(f)
            sig = bandpass_filter(raw, fs=FS, low=BANDPASS[0], high=BANDPASS[1])
            sig = clip_artifacts(sig, THRESH_UV)
            sig = zscore_per_channel(sig)
            epo = segment_epochs(sig, fs=FS)  # (E, T, C)
            if len(epo) == 0: continue
            Xs.append(epo)
            ys.append(np.full((epo.shape[0],), cls_label, dtype=np.int32))
            gid = os.path.basename(f)
            gs.append(np.array([gid]*epo.shape[0], dtype=object))
    if not Xs: raise RuntimeError("Keine Daten gefunden – Pfad/Endungen prüfen.")
    X = np.vstack(Xs).astype(np.float32)   # (N, T, C)
    y = np.concatenate(ys).astype(np.int32)
    g = np.concatenate(gs)
    return X, y, g

def group_split(X, y, groups, train_size=0.70, seed=SEED):
    gss = GroupShuffleSplit(n_splits=1, train_size=train_size, random_state=seed)
    tr_idx, tmp_idx = next(gss.split(X, y, groups))
    X_tr, y_tr, g_tr = X[tr_idx], y[tr_idx], groups[tr_idx]
    X_tmp, y_tmp, g_tmp = X[tmp_idx], y[tmp_idx], groups[tmp_idx]
    gss2 = GroupShuffleSplit(n_splits=1, train_size=0.5, random_state=seed)
    va_idx, te_idx = next(gss2.split(X_tmp, y_tmp, g_tmp))
    return X_tr, y_tr, X_tmp[va_idx], y_tmp[va_idx], X_tmp[te_idx], y_tmp[te_idx]

# Zelle 6: tf.data Pipelines (inkl. STFT-Mapping) – stabil (repeat + steps)
def mild_noise_tf(a, b=None, std=0.01, p=0.3):
    def _noisy(z): return z + tf.random.normal(tf.shape(z), stddev=std, dtype=z.dtype)
    if b is None:
        return tf.cond(tf.random.uniform([]) < p, lambda: _noisy(a), lambda: a)
    else:
        a2 = tf.cond(tf.random.uniform([]) < p, lambda: _noisy(a), lambda: a)
        return a2, b

def prep_ds(X, y=None, training=False, batch=BATCH):
    if y is None:
        ds = tf.data.Dataset.from_tensor_slices((X,))
    else:
        ds = tf.data.Dataset.from_tensor_slices((X, y))
    if training:
        ds = ds.shuffle(min(len(X), 20000), seed=SEED, reshuffle_each_iteration=True)
        ds = ds.map(mild_noise_tf, num_parallel_calls=AUTO)
    # STFT + Log
    if y is None:
        ds = ds.map(lambda a: (to_log_stft(a),), num_parallel_calls=AUTO)
    else:
        ds = ds.map(lambda a,b: (to_log_stft(a), b), num_parallel_calls=AUTO)
    if training:
        ds = ds.repeat()  # stabil: keine leeren Epochen
    ds = ds.batch(batch).prefetch(AUTO)
    return ds

# Zelle 7: 2D-CNN (Zeit x Frequenz, Kanäle als Depth)
def build_spec_cnn(dropout=0.35):
    # Input-Shape dynamisch aus Dataset
    ex = next(iter(ds_val))[0]
    H, W, C = ex.shape[1], ex.shape[2], ex.shape[3]   # (frames, kept_bins, C)
    inp = layers.Input(shape=(H, W, C))

    # Block 1
    x = layers.Conv2D(32, (3,3), padding="same", use_bias=False)(inp)
    x = layers.BatchNormalization()(x); x = layers.ReLU()(x)
    x = layers.MaxPool2D((2,2))(x)             # ↓ H//2, W//2

    # Block 2  (⚠️ Pool nur über Frequenz, weil H nach Block 1 evtl. schon 1 ist)
    x = layers.Conv2D(64, (3,3), padding="same", use_bias=False)(x)
    x = layers.BatchNormalization()(x); x = layers.ReLU()(x)
    x = layers.MaxPool2D((1,2))(x)             # ← Fix: nur Frequenz halbieren

    # Block 3 (ohne weiteres Pooling)
    x = layers.Conv2D(128, (3,3), padding="same", use_bias=False)(x)
    x = layers.BatchNormalization()(x); x = layers.ReLU()(x)

    x = layers.GlobalAveragePooling2D()(x)
    x = layers.Dense(128, use_bias=False)(x)
    x = layers.BatchNormalization()(x); x = layers.ReLU()(x)
    x = layers.Dropout(dropout)(x)

    out = layers.Dense(1, activation="sigmoid")(x)
    return models.Model(inp, out)
