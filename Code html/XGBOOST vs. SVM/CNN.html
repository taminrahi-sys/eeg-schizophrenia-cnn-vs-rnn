# Zelle 2: Loader & Vorverarbeitung

def load_eea_file(path):
    """
    >>> HIER deinen echten .eea-Parser implementieren. <<<
    Erwartet: np.ndarray mit shape (n_channels, n_samples), dtype float
    Platzhalter unten erzeugt deterministische Dummy-Daten, damit das Notebook läuft.
    """
    rng = np.random.default_rng(abs(hash(path)) % (2**32))
    n_channels, n_samples = 32, 10_000
    return rng.normal(0, 1, size=(n_channels, n_samples)).astype(np.float32)

def bandpass_filter(eeg, fs=FS, low=1, high=40, order=4):
    """Butterworth-Bandpass (kanalweise) für (C, T)."""
    ny = 0.5 * fs
    b, a = butter(order, [low/ny, high/ny], btype='band')
    return filtfilt(b, a, eeg, axis=1)

def zscore_per_channel(eeg, eps=1e-8):
    """Z-Score pro Kanal (C, T)."""
    m = eeg.mean(axis=1, keepdims=True)
    s = eeg.std(axis=1, keepdims=True)
    return (eeg - m) / (s + eps)

def clip_artifacts(eeg, thresh_uv=THRESH_UV):
    """Sanftes Clipping großer Ausreißer (optional)."""
    if thresh_uv is None:
        return eeg
    return np.clip(eeg, -thresh_uv, thresh_uv)

def segment_epochs(eeg, fs=FS, epoch_len_s=EPOCH_LEN_S, step_s=STEP_S):
    """
    Zerlegt (C, T) in (E, T_win, C) für Keras-1D-CNN (timesteps, features).
    """
    win = int(epoch_len_s * fs)
    step = int(step_s * fs)
    C, T = eeg.shape
    if T < win:
        return np.empty((0, win, C), dtype=eeg.dtype)

    epochs = []
    for start in range(0, T - win + 1, step):
        seg = eeg[:, start:start+win]          # (C, win)
        seg = np.transpose(seg, (1, 0))        # -> (win, C)
        epochs.append(seg)
    return np.stack(epochs, axis=0)            # (E, win, C)

# Zelle 3: Datensatz aufbauen (Load -> Vorverarbeitung -> Segmentierung)
def build_dataset(data_dir):
    Xs, ys = [], []
    for cls_name, cls_label in CLASS_MAP.items():
        folder = os.path.join(data_dir, cls_name)
        files = sorted(glob.glob(os.path.join(folder, "*.eea")))
        for f in files:
            raw = load_eea_file(f)                                      # (C, T)
            sig = bandpass_filter(raw, fs=FS, low=BANDPASS[0], high=BANDPASS[1])
            sig = clip_artifacts(sig, THRESH_UV)
            sig = zscore_per_channel(sig)
            epochs = segment_epochs(sig, fs=FS)                         # (E, win, C)
            if epochs.shape[0] == 0:
                continue
            Xs.append(epochs)
            ys.append(np.full((epochs.shape[0],), cls_label, dtype=int))

    if not Xs:
        raise RuntimeError("Keine Daten gefunden. Pfad/Endungen prüfen.")
    X = np.vstack(Xs)                 # (N, win, C)
    y = np.concatenate(ys)            # (N,)
    return X.astype(np.float32), y.astype(np.int32)

# Zelle 5: tf.data Pipelines + einfache Augmentation
BATCH = 64
AUTO = tf.data.AUTOTUNE

def add_noise(x, std=0.02):
    noise = tf.random.normal(tf.shape(x), stddev=std, dtype=x.dtype)
    return x + noise

def ds_from_numpy(X, y=None, training=False):
    if y is None:
        ds = tf.data.Dataset.from_tensor_slices((X,))
    else:
        ds = tf.data.Dataset.from_tensor_slices((X, y))
    if training:
        ds = ds.shuffle(min(len(X), 10_000), seed=SEED, reshuffle_each_iteration=True)
        ds = ds.map(lambda a, b: (add_noise(a), b), num_parallel_calls=AUTO)
    ds = ds.batch(BATCH).prefetch(AUTO)
    return ds

ds_train = ds_from_numpy(X_train, y_train, training=True)
ds_val   = ds_from_numpy(X_val,   y_val,   training=False)

# Zelle 6: 1D-CNN Modell definieren
def build_cnn(input_len=WIN, n_ch=N_CH):
    inp = layers.Input(shape=(input_len, n_ch))  # (timesteps, features=channels)

    x = layers.Conv1D(64, 7, padding="same")(inp)
    x = layers.BatchNormalization()(x)
    x = layers.ReLU()(x)
    x = layers.Dropout(0.2)(x)

    x = layers.Conv1D(64, 7, padding="same")(x)
    x = layers.BatchNormalization()(x)
    x = layers.ReLU()(x)
    x = layers.MaxPool1D(2)(x)

    x = layers.Conv1D(128, 5, padding="same")(x)
    x = layers.BatchNormalization()(x)
    x = layers.ReLU()(x)
    x = layers.Dropout(0.3)(x)

    x = layers.Conv1D(128, 5, padding="same")(x)
    x = layers.BatchNormalization()(x)
    x = layers.ReLU()(x)
    x = layers.MaxPool1D(2)(x)

    x = layers.Conv1D(256, 3, padding="same")(x)
    x = layers.BatchNormalization()(x)
    x = layers.ReLU()(x)
    x = layers.GlobalAveragePooling1D()(x)

    x = layers.Dense(128)(x)
    x = layers.BatchNormalization()(x)
    x = layers.ReLU()(x)
    x = layers.Dropout(0.4)(x)

    out = layers.Dense(1, activation="sigmoid")(x)
    model = models.Model(inp, out)
    return model

model = build_cnn()
model.summary()
