Schizophrenie vs. Normal – RNN-Klassifikation¶

Dieses Notebook lädt .eea-Dateien aus den Ordnern norm und sch, führt eine Vorverarbeitung durch, erstellt Sequenzen für ein RNN (LSTM/GRU), trainiert ein binäres Modell und gibt Accuracy, F1, Precision, Recall und die Konfusionsmatrix aus.

Hinweise:

  • Der .eea-Parser verwendet heuristische Laderoutinen (Text/CSV/Binär float32). Passen Sie parse_eea an, falls Ihr .eea-Format proprietär ist.
  • Die Sequenzlänge (WINDOW_SIZE) und Schrittweite (STEP_SIZE) können je nach Samplingrate angepasst werden.
  • Es wird ein dateibasierter Split vorgenommen (keine Leckage zwischen Train/Val/Test).

# Loader/Parser für .eea-Dateien mit Heuristiken
def try_load_text_csv(path: Path):
    # Versuch: whitespace-getrennt
    try:
        arr = np.loadtxt(path)
        return arr
    except Exception:
        pass
    # Versuch: CSV mit Komma
    try:
        arr = pd.read_csv(path, header=None).values
        return arr
    except Exception:
        pass
    # Versuch: CSV mit Semikolon
    try:
        arr = pd.read_csv(path, header=None, sep=';').values
        return arr
    except Exception:
        pass
    return None

def try_load_binary_float32(path: Path):
    try:
        data = np.fromfile(path, dtype=np.float32)
        if data.size == 0:
            return None
        return data
    except Exception:
        return None

def ensure_2d(arr: np.ndarray) -> np.ndarray:
    arr = np.asarray(arr)
    if arr.ndim == 1:
        arr = arr.reshape(-1, 1)
    return arr

def clean_array(arr: np.ndarray) -> np.ndarray:
    arr = np.asarray(arr, dtype=np.float32)
    # Entferne NaN/Inf Zeilen
    mask = np.all(np.isfinite(arr), axis=1)
    arr = arr[mask]
    # Optional leichte Enttrendung
    try:
        arr = detrend(arr, axis=0, type='linear')
    except Exception:
        pass
    # Clipping extremer Ausreißer (robust)
    q_low, q_high = np.nanpercentile(arr, [0.5, 99.5], axis=0)
    arr = np.clip(arr, q_low, q_high)
    return arr.astype(np.float32)

def parse_eea(path: Path) -> np.ndarray:
    # 1) Text/CSV
    arr = try_load_text_csv(path)
    # 2) Binär float32 ohne Header
    if arr is None:
        arr = try_load_binary_float32(path)
    if arr is None:
        raise ValueError(f'Konnte Datei nicht laden: {path}')
    arr = ensure_2d(arr)
    arr = clean_array(arr)
    return arr

def segment_array(arr: np.ndarray, window_size: int, step: int, max_segments: int | None = None):
    n = arr.shape[0]
    if n < window_size:
        return np.empty((0, window_size, arr.shape[1]), dtype=np.float32)
    starts = np.arange(0, n - window_size + 1, step)
    if max_segments is not None:
        # Zufällige Auswahl zur Limitierung
        rng = np.random.default_rng(SEED)
        if starts.size > max_segments:
            starts = rng.choice(starts, size=max_segments, replace=False)
        starts = np.sort(starts)
    segments = np.stack([arr[s:s+window_size, :] for s in starts], axis=0)
    return segments.astype(np.float32)

def compute_channel_stats(X: np.ndarray):
    # X: (N, T, C) -> Mittel/Std pro Kanal über alle N*T Werte
    C = X.shape[-1]
    means = X.reshape(-1, C).mean(axis=0)
    stds  = X.reshape(-1, C).std(axis=0) + 1e-8
    return means, stds

def apply_channel_norm(X: np.ndarray, means: np.ndarray, stds: np.ndarray):
    return (X - means) / stds

def stack_sets(idxs):
    X_list, y_list = [], []
    for i in idxs:
        X_list.append(per_file_data[i]['X'])
        y_list.append(np.full(per_file_data[i]['X'].shape[0], per_file_data[i]['label'], dtype=np.int64))

    y = np.concatenate(y_list, axis=0)
    return X, y

# Schnelleres Modell: Conv1D-Feature-Extractor + GRU
def build_model(input_shape):
    inp = layers.Input(shape=input_shape)
    x = layers.Conv1D(64, 7, strides=2, padding='same', activation='relu')(inp)
    x = layers.BatchNormalization()(x)
    x = layers.Conv1D(128, 5, strides=2, padding='same', activation='relu')(x)
    x = layers.BatchNormalization()(x)
    x = layers.SpatialDropout1D(0.2)(x)
    x = layers.Bidirectional(layers.GRU(64))(x)
    x = layers.Dropout(0.3)(x)
    x = layers.Dense(128, activation='relu')(x)
    x = layers.Dropout(0.2)(x)
    out = layers.Dense(1, activation='sigmoid', dtype='float32')(x)  # float32-Ausgabe für stabilen BCE
    model = models.Model(inp, out)
    model.compile(
        optimizer=tf.keras.optimizers.Adam(1e-3),
        loss='binary_crossentropy',
        metrics=[
            'accuracy',
            tf.keras.metrics.AUC(name='auc'),
            tf.keras.metrics.Precision(name='prec'),
            tf.keras.metrics.Recall(name='rec')
        ]
    )
    return model

input_shape = (X_train.shape[1], X_train.shape[2])
model = build_model(input_shape)
model.summary()

# Klassenungleichgewicht berücksichtigen
from collections import Counter
cnt = Counter(y_train.tolist())
total = len(y_train)
class_weight = {c: total/(2.0*cnt[c]) for c in cnt}
print('Class weights:', class_weight)

# Sample-Weights in tf.data einspeisen (class_weight mit Dataset)
cw0 = tf.cast(class_weight.get(0, 1.0), tf.float32)
cw1 = tf.cast(class_weight.get(1, 1.0), tf.float32)
def add_weights(x, y):
    w = tf.where(tf.equal(tf.cast(y, tf.int32), 1), cw1, cw0)
    return x, y, w
train_ds_w = train_ds.map(add_weights, num_parallel_calls=AUTOTUNE)

cbs = [
    callbacks.EarlyStopping(monitor='val_f1', mode='max', patience=PATIENCE, restore_best_weights=True),
    callbacks.ReduceLROnPlateau(monitor='val_f1', mode='max', factor=0.5, patience=max(4, PATIENCE//3), min_lr=1e-5)
]

# Custom F1-Metrik für Monitoring (nur für Callback/Logs)
# Hinweis: Für das Training bleibt der Loss BCE, F1 nur zur Früherkennung
class F1ScoreCallback(callbacks.Callback):
    def __init__(self, x_val, y_val):
        super().__init__()
        self.xv, self.yv = x_val, y_val
    def on_epoch_end(self, epoch, logs=None):
        y_prob = self.model.predict(self.xv, verbose=0).ravel()
        y_pred = (y_prob >= 0.5).astype(int)
        f1 = f1_score(self.yv, y_pred)
        if logs is not None:
            logs['val_f1'] = f1
        print(f' - val_f1: {f1:.4f}', end='')

f1_cb = F1ScoreCallback(val_ds, y_val)
history = model.fit(
    train_ds_w,
    validation_data=val_ds,
    epochs=EPOCHS,
    callbacks=[f1_cb] + cbs,
    verbose=1
)

# Schwellenwert-Tuning auf Val zur Verbesserung von F1
y_val_prob = model.predict(val_ds, verbose=0).ravel()
best_thr, best_f1 = 0.5, -1.0
for thr in np.linspace(0.1, 0.9, 81):
    f1 = f1_score(y_val, (y_val_prob >= thr).astype(int))
    if f1 > best_f1:
        best_f1, best_thr = f1, thr
print(f'Gewählter Schwellenwert (Val-basiert): {best_thr:.3f} mit F1={best_f1:.4f}')
